-- ============================================================
-- ESPORTS SCHOOL LEAGUE DATABASE — Supabase (Postgres)
-- Multi-game, multi-season, handles student turnover.
-- Run in: Supabase Dashboard > SQL Editor > New Query
-- ============================================================

-- ------------------------------------------------------------
-- CORE ENTITIES
-- ------------------------------------------------------------

CREATE TABLE games (
    game_id     BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name        TEXT NOT NULL UNIQUE,
    is_active   BOOLEAN NOT NULL DEFAULT TRUE
);

CREATE TABLE schools (
    school_id   BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name        TEXT NOT NULL UNIQUE,
    joined_at   DATE DEFAULT CURRENT_DATE,
    is_active   BOOLEAN NOT NULL DEFAULT TRUE
);

CREATE TABLE people (
    person_id       BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    first_name      TEXT NOT NULL,
    last_name       TEXT NOT NULL,
    email           TEXT,
    discord         TEXT,
    graduation_year INT,
    school_id       BIGINT REFERENCES schools(school_id) ON DELETE RESTRICT,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- game_id nullable: NULL = league-wide season, set = game-specific season
CREATE TABLE seasons (
    season_id   BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    game_id     BIGINT REFERENCES games(game_id) ON DELETE RESTRICT,
    name        TEXT NOT NULL,
    start_date  DATE,
    end_date    DATE,
    is_active   BOOLEAN NOT NULL DEFAULT FALSE
);

-- ------------------------------------------------------------
-- TEAMS & ROSTERS
-- ------------------------------------------------------------

CREATE TABLE teams (
    team_id     BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    school_id   BIGINT NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
    game_id     BIGINT NOT NULL REFERENCES games(game_id) ON DELETE RESTRICT,
    season_id   BIGINT NOT NULL REFERENCES seasons(season_id) ON DELETE RESTRICT,
    division    TEXT NOT NULL CHECK (division IN ('A', 'B')),
    UNIQUE (school_id, game_id, season_id, division)
);

-- Players on a specific team. One person can be on multiple teams
-- across different games/seasons.
CREATE TABLE players (
    player_id       BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    team_id         BIGINT NOT NULL REFERENCES teams(team_id) ON DELETE CASCADE,
    person_id       BIGINT NOT NULL REFERENCES people(person_id) ON DELETE RESTRICT,
    ign             TEXT,
    tracker_url     TEXT,
    role            TEXT,                   -- in-game role (e.g. Duelist, ADC)
    is_substitute   BOOLEAN NOT NULL DEFAULT FALSE,
    is_captain      BOOLEAN NOT NULL DEFAULT FALSE,
    joined_date     DATE DEFAULT CURRENT_DATE,
    left_date       DATE,
    UNIQUE (team_id, person_id)
);

-- Team/school-level staff. team_id is set for team-level staff
-- (manager, coach), NULL for school-wide roles (club president).
-- A person can be both a player and team_staff simultaneously.
CREATE TABLE team_staff (
    team_staff_id   BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    person_id       BIGINT NOT NULL REFERENCES people(person_id) ON DELETE RESTRICT,
    team_id         BIGINT REFERENCES teams(team_id) ON DELETE SET NULL,
    school_id       BIGINT NOT NULL REFERENCES schools(school_id) ON DELETE RESTRICT,
    season_id       BIGINT NOT NULL REFERENCES seasons(season_id) ON DELETE RESTRICT,
    title           TEXT NOT NULL,          -- e.g. president, manager, coach
    UNIQUE (person_id, team_id, season_id, title)
);

-- Organization-level staff who run the league itself.
-- Not tied to any school or team.
CREATE TABLE org_staff (
    org_staff_id    BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    person_id       BIGINT NOT NULL REFERENCES people(person_id) ON DELETE RESTRICT,
    title           TEXT NOT NULL,          -- e.g. commissioner, admin, organizer
    start_date      DATE DEFAULT CURRENT_DATE,
    end_date        DATE,                   -- NULL = currently active
    UNIQUE (person_id, title)
);

-- ------------------------------------------------------------
-- MATCHES
-- ------------------------------------------------------------

CREATE TABLE matches (
    match_id        BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    season_id       BIGINT NOT NULL REFERENCES seasons(season_id) ON DELETE CASCADE,
    game_id         BIGINT NOT NULL REFERENCES games(game_id) ON DELETE RESTRICT,
    match_date      DATE,
    team_home_id    BIGINT NOT NULL REFERENCES teams(team_id) ON DELETE RESTRICT,
    team_away_id    BIGINT NOT NULL REFERENCES teams(team_id) ON DELETE RESTRICT,
    home_score      INT DEFAULT 0,
    away_score      INT DEFAULT 0,
    mvp_person_id   BIGINT REFERENCES people(person_id) ON DELETE SET NULL,
    match_details   JSONB,                 -- game-specific breakdown (maps, rounds, etc.), nullable
    status          TEXT NOT NULL DEFAULT 'scheduled'
                    CHECK (status IN ('scheduled','completed','forfeit','postponed','cancelled')),
    notes           TEXT,
    CHECK (team_home_id != team_away_id)
);

CREATE INDEX idx_people_school        ON people(school_id);
CREATE INDEX idx_people_grad_year     ON people(graduation_year);
CREATE INDEX idx_teams_school_season  ON teams(school_id, season_id);
CREATE INDEX idx_teams_game           ON teams(game_id);
CREATE INDEX idx_players_team          ON players(team_id);
CREATE INDEX idx_players_person        ON players(person_id);
CREATE INDEX idx_staff_person          ON team_staff(person_id);
CREATE INDEX idx_staff_school_season   ON team_staff(school_id, season_id);
CREATE INDEX idx_org_staff_person      ON org_staff(person_id);
CREATE INDEX idx_matches_season       ON matches(season_id);
CREATE INDEX idx_matches_game         ON matches(game_id);
CREATE INDEX idx_matches_teams        ON matches(team_home_id, team_away_id);
CREATE INDEX idx_matches_details       ON matches USING GIN (match_details);
CREATE INDEX idx_matches_mvp          ON matches(mvp_person_id);

CREATE VIEW v_roster AS
SELECT
    pl.player_id,
    p.first_name,
    p.last_name,
    p.graduation_year,
    p.discord,
    pl.ign,
    pl.tracker_url,
    pl.role,
    pl.is_substitute,
    pl.is_captain,
    t.division AS team_division,
    s.name AS school_name,
    g.name AS game_name,
    se.name AS season_name
FROM players pl
JOIN people p   ON pl.person_id = p.person_id
JOIN teams t    ON pl.team_id = t.team_id
JOIN schools s  ON t.school_id = s.school_id
JOIN games g    ON t.game_id = g.game_id
JOIN seasons se ON t.season_id = se.season_id
WHERE pl.left_date IS NULL;

-- Standings: computed live from matches.
-- Calculates match/map wins and losses from completed matches.
-- Round-level data lives in map_details JSONB and is game-specific,
-- so it is not computed here. Query map_details directly for that.
CREATE VIEW v_standings AS
SELECT
    t.team_id,
    s.name AS school_name,
    g.name AS game_name,
    t.division AS team_division,
    se.name AS season_name,
    COALESCE(SUM(CASE
        WHEN (m.team_home_id = t.team_id AND m.home_score > m.away_score)
          OR (m.team_away_id = t.team_id AND m.away_score > m.home_score)
        THEN 1 ELSE 0 END), 0) AS match_wins,
    COALESCE(SUM(CASE
        WHEN (m.team_home_id = t.team_id AND m.home_score < m.away_score)
          OR (m.team_away_id = t.team_id AND m.away_score < m.home_score)
        THEN 1 ELSE 0 END), 0) AS match_losses,
    COALESCE(SUM(CASE WHEN m.team_home_id = t.team_id THEN m.home_score
                      WHEN m.team_away_id = t.team_id THEN m.away_score
                      ELSE 0 END), 0) AS sub_wins,
    COALESCE(SUM(CASE WHEN m.team_home_id = t.team_id THEN m.away_score
                      WHEN m.team_away_id = t.team_id THEN m.home_score
                      ELSE 0 END), 0) AS sub_losses
FROM teams t
JOIN schools s  ON t.school_id = s.school_id
JOIN games g    ON t.game_id = g.game_id
JOIN seasons se ON t.season_id = se.season_id
LEFT JOIN matches m
    ON m.status = 'completed'
    AND (m.team_home_id = t.team_id OR m.team_away_id = t.team_id)
GROUP BY t.team_id, s.name, g.name, t.division, se.name
ORDER BY match_wins DESC, sub_wins DESC;

CREATE VIEW v_match_results AS
SELECT
    m.match_id,
    g.name AS game_name,
    se.name AS season_name,
    m.match_date,
    sh.name AS home_school,
    th.division  AS home_division,
    sa.name AS away_school,
    ta.division  AS away_division,
    m.home_score,
    m.away_score,
    m.match_details,
    m.status,
    m.notes,
    pm.first_name || ' ' || pm.last_name AS mvp_name
FROM matches m
JOIN games g        ON m.game_id = g.game_id
JOIN seasons se     ON m.season_id = se.season_id
JOIN teams th       ON m.team_home_id = th.team_id
JOIN schools sh     ON th.school_id = sh.school_id
JOIN teams ta       ON m.team_away_id = ta.team_id
JOIN schools sa     ON ta.school_id = sa.school_id
LEFT JOIN people pm ON m.mvp_person_id = pm.person_id;

-- Team/school staff with context
CREATE VIEW v_team_staff AS
SELECT
    ts.team_staff_id,
    p.first_name,
    p.last_name,
    p.graduation_year,
    p.discord,
    ts.title,
    s.name AS school_name,
    t.division AS team_division,
    g.name AS game_name,
    se.name AS season_name
FROM team_staff ts
JOIN people p    ON ts.person_id = p.person_id
JOIN schools s   ON ts.school_id = s.school_id
JOIN seasons se  ON ts.season_id = se.season_id
LEFT JOIN teams t ON ts.team_id = t.team_id
LEFT JOIN games g ON t.game_id = g.game_id;

-- Org staff: active league-level staff only
CREATE VIEW v_org_staff AS
SELECT
    os.org_staff_id,
    p.first_name,
    p.last_name,
    p.discord,
    p.email,
    os.title,
    os.start_date
FROM org_staff os
JOIN people p ON os.person_id = p.person_id
WHERE os.end_date IS NULL;

-- Unified person overview: shows every role a person holds across the league
CREATE VIEW v_people AS
SELECT
    p.person_id,
    p.first_name,
    p.last_name,
    p.email,
    p.discord,
    p.graduation_year,
    s.name AS school_name,
    COALESCE(
        (SELECT json_agg(json_build_object(
            'game', g2.name,
            'division', t2.division,
            'role', pl.role,
            'is_captain', pl.is_captain,
            'season', se2.name
        ))
        FROM players pl
        JOIN teams t2 ON pl.team_id = t2.team_id
        JOIN games g2 ON t2.game_id = g2.game_id
        JOIN seasons se2 ON t2.season_id = se2.season_id
        WHERE pl.person_id = p.person_id AND pl.left_date IS NULL),
        '[]'::json
    ) AS active_player_roles,
    COALESCE(
        (SELECT json_agg(json_build_object(
            'title', ts.title,
            'school', s2.name,
            'season', se3.name
        ))
        FROM team_staff ts
        JOIN schools s2 ON ts.school_id = s2.school_id
        JOIN seasons se3 ON ts.season_id = se3.season_id
        WHERE ts.person_id = p.person_id),
        '[]'::json
    ) AS team_staff_roles,
    COALESCE(
        (SELECT json_agg(json_build_object(
            'title', os.title,
            'start_date', os.start_date
        ))
        FROM org_staff os
        WHERE os.person_id = p.person_id AND os.end_date IS NULL),
        '[]'::json
    ) AS org_staff_roles
FROM people p
LEFT JOIN schools s ON p.school_id = s.school_id;
